package cn.jqe.hive;
import org.apache.hadoop.hive.ql.exec.UDF;
//定义Java类hiveUDF，并继承UDF类
public class hiveUDF extends UDF {
	//重写evaluate()方法
	public String evaluate(double column1, double column2) {
		if (column1 > column2) {
			return "min:"+column2+",diffe:"+(column1-column2);
		}else if (column1<column2){
			return "min:"+column1+",diffe:"+(column2-column1);
		}else return "equal";
	}
}
